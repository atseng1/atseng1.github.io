Bar Chart

  count(boro) %>% 
  mutate(boro = fct_reorder(boro, n)) %>% 
  plot_ly(x = ~boro, y = ~n, color = ~boro, type = "bar") %>% 
  layout(
    title = "Number of Restaurant Inspections Across New York City",

Violin Plot

  rest_inspec %>% 
  inner_join(top_cuisines, by = "cuisine_description") %>% 
  mutate(
    cuisine_description = fct_reorder(cuisine_description, score),
    text_label = str_c("Cuisine Type: ", cuisine_description, '\nScore: ', score)) %>%
  plot_ly(
    x = ~boro, y = ~score, type = "violin", mode = "markers",
    color = ~boro, text = ~text_label, alpha = 0.5) %>% 
  layout(
    title = "Restaurant Inspection Scores of the Top 8 Cuisines In New York City",
    xaxis = list(title = "Borough"), 
    yaxis = list(title = "Restaurant Inspection Score"))
